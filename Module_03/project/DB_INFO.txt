CREATE DATABASE pizza;

USE pizza;
CREATE TABLE menu (id INT AUTO_INCREMENT PRIMARY KEY,
                   name varchar(255),
                   ingredients varchar(255),
                   price SMALLINT DEFAULT NULL);
